SetMenuItemText()

Syntaxe

SetMenuItemText(#Menu, Element, Texte$)
Description
Change le texte d'un élément d'un menu.

Arguments

#Menu Le menu à utiliser.
Element Le numéro de l'élément.
Texte$ Le nouveau texte.

Valeur de retour

Aucune.

Exemple

  If OpenWindow(0, 200, 200, 300, 100, "Exemple  SetMenuItemText")
    ButtonGadget(0,70,10,150,30,"Changer le menu Ouvrir")
    If CreateMenu(0, WindowID(0))
      MenuTitle("Projet")
      MenuItem(1, "Ouvrir")     
    EndIf
    
    Repeat
      Event = WaitWindowEvent()          
      If Event = #PB_Event_Gadget  
        Select EventGadget()
          Case 0 
           SetMenuItemText(0,1,"Nouveau") ; Change le menu "Ouvrir" par "Nouveau" 
        EndSelect
      EndIf
    Until Event = #PB_Event_CloseWindow
  EndIf
